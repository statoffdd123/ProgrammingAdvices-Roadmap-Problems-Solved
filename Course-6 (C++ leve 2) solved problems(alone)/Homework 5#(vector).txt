#include <iostream>
#include <vector>

std::vector<int> ReadNumber(std::vector<int>& Numbers)
{
    int Number = 0;

    do
    {
        std::printf("Please Enter The Number You Want?\n");
        std::cin >> Number;
        Numbers.push_back(Number);
        std::printf("\ndo you want to add more? y/n: ");

}
void PrintNumbers(std::vector<int>& Numbers)
{

}
