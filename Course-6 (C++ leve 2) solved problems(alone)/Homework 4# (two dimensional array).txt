#include <iostream>


int main()
{
	int Led_Zeppelin[10][10];

	
	for (int i = 0; i < 10 ;i++)
	{

		for (int x = 0; x < 10; x++)
		{
			Led_Zeppelin[x][i] = (x+1) * (i+1);
		}		

	}

	for (int i = 0; i < 10; i++)
	{

		for (int x = 0; x < 10; x++)
		{
			printf("%0*d",2, Led_Zeppelin[x][i]); std::cout << "  ";
		}
		std::cout << "\n";
	}

	
	return 0;
}
